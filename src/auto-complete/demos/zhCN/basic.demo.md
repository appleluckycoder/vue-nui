# 基础用法

```html
<n-auto-complete :options="options" v-model:value="value" placeholder="邮箱" />
```

```js
export default {
  computed: {
    options () {
      return ['@gmail.com', '@163.com', '@qq.com'].map((suffix) => {
        const prefix = this.value.split('@')[0]
        return {
          label: prefix + suffix,
          value: prefix + suffix
        }
      })
    }
  },
  data () {
    return {
      value: ''
    }
  }
}
```
